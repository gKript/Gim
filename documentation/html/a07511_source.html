76 _gim_flag gim_list_obj::del_item( _gim_Uint32 index ) {
77  if ( index = 0 ) {
78  gim_error->set( GIM_ERROR_WARNING , "gim_list_obj::del_item" , "index cannot be 0" , __GIM_ERROR );
79  return __GIM_NOT_OK;
80  }
81 
82  _gim_list_item * tmplist = startlist;
83  _gim_list_item * beforelist = NULL;
84  _gim_list_item * afterlist = NULL;
85 
86  if( index == 0 ) {
87  startlist = startlist->nextitem;
88  currentlist = startlist;
89  gim_error->set( "gim_list_obj::add_item" , "Item deleted" );
90  }
91 
92  while ( tmplist != NULL ) {
93  if ( tmplist->index == index ) {
94  beforelist = tmplist->previtem;
95  afterlist = tmplist->nextitem;
96  beforelist->nextitem = afterlist;
97  afterlist->previtem = beforelist;
98  gim_memory->Unlock_and_free ( tmplist );
99  item_counter--;
100  gim_error->set( "gim_list_obj::add_item" , "Item found and deleted" );
101  return __GIM_OK;
102  }
103  tmplist = tmplist->nextitem;
104  }
105  gim_error->set( GIM_ERROR_WARNING , "gim_list_obj::del_item" , "Item NOT FOUND" , __GIM_ERROR );
106  return __GIM_NOT_OK;
107 }

110 void gim_list_obj::destroy_list( void ) {
111  if ( startlist == NULL )
112  gim_error->Set( GIM_ERROR_WARNING , "gim_list_obj::destroy_list" , "The list is already empty." );
113  else {
114 
115  _gim_list_item * temp_item = startlist;
116  _gim_list_item * temp_free;
117  _gim_Uint16 It = 0 , Ls = 0;
118 
119 
120  while ( temp_item != NULL ) {
121  if ( temp_item->item != NULL ) {
122  if ( gim_memory->Check ( temp_item->item ) == __GIM_EXIST ) {
123  if ( gim_memory->Free ( temp_item->item ) == __GIM_NOT_OK )
124  gim_error->set( GIM_ERROR_CRITICAL , "gim_list_obj::destroy_list" , "Item NOT found. You have to deallocate it before to destroy the list" , __GIM_ERROR );
125  else
126  It++;
127  }
128  else
129  gim_error->Set( GIM_ERROR_WARNING , "gim_list_obj::destroy_list" , "Item not found" );
130  }
131  else
132  gim_error->Set( GIM_ERROR_WARNING , "gim_list_obj::destroy_list" , "Item was NULL" );
133  temp_free = temp_item;
134  temp_item = temp_item->nextitem;
135  if ( gim_memory->Free ( temp_free ) == __GIM_NOT_OK )
136  gim_error->set( GIM_ERROR_CRITICAL , "gim_list_obj::destroy_list" , "Some problem to deallocate a List item." , __GIM_ERROR );
137  else
138  Ls++;
139  }
140  gim_error->Set( GIM_ERROR_MESSAGE , "gim_list_obj::destroy_list" , "Deallocated : [%3d List items] [%3d Linked items]" , Ls , It );
141  }
142 }
