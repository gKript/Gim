161 void gim_password_obj::calc( void ) {
162  _gim_Uint8 c = 0 , t=0 ;
163  gim_error->set( "gim_password_obj::calc" , "Calculating passworg digest and seed..." );
164  if ( strlen( password ) == 0 )
165  gim_error->set( GIM_ERROR_CRITICAL , "gim_password_obj::calc" , "No password setted. Security issue?" , __GIM_ERROR );
166  if ( password_already_calc == __GIM_NO ) {
167 // puts("Calcolo!!!");
168  t = (_gim_int8)strlen( password );
169  seed_md5 = 0 ;
170  seed_sha1 = 0 ;
171  seed_sha256 = 0 ;
172  seed_sha512 = 0 ;
173  seed_secondary = 0;
174  seed_main = 1 ;
175  // init - Digest
176 // puts("Copio i digest nella struttura");
177  strncpy(password_md5 , gim_checksum->md5 ( password , (unsigned long)t - 1 ) , sizeof(password_md5) );
178  strncpy(password_sha1 , gim_checksum->sha1 ( password_md5 , (unsigned long)strlen( password_md5 ) - 1 ) , sizeof(password_sha1) );
179  strncpy(password_sha256 , gim_checksum->sha256( password_sha1 , (unsigned long)strlen( password_sha1 ) - 1 ) , sizeof(password_sha256) );
180  strncpy(password_sha512 , gim_checksum->sha512( password_sha256 , (unsigned long)strlen( password_sha256 ) - 1 ) , sizeof(password_sha512) );
181 // puts("Fatto... mi occupo della secondary");
182  if ( strlen( secondary ) )
183  strcpy(secondary_sha512 , gim_checksum->sha512( secondary , (unsigned long)strlen( secondary ) ) );
184  // init - Sha1 Seed
185 // puts("Fatto... un po' di calcoli sui digest");
186  for ( c = 0 ; c < GIM_SHA1_SIZE ; c++ )
187  seed_sha1 += password_sha1[c] * Prime[c];
188  // init - Md5 Seed
189  for ( c = 0 ; c < GIM_MD5_SIZE ; c++ )
190  seed_md5 += password_md5[c] * Prime[c];
191  // init - Sha256 Seed
192  for ( c = 0 ; c < GIM_SHA256_SIZE ; c++ )
193  seed_sha256 += password_sha256[c] * Prime[c];
194  // init - Sha512 Seed
195  for ( c = 0 ; c < GIM_SHA512_SIZE ; c++ )
196  seed_sha512 += password_sha512[c] * Prime[c];
197  if ( strlen( secondary_sha512 ) ) {
198  for ( c = 0 ; c < GIM_SHA512_SIZE ; c++ )
199  seed_secondary += secondary_sha512[c] * Prime[c];
200  }
201 // puts("Fatto... altri calcoli");
202  seed_main *= seed_md5;
203  seed_main *= seed_sha1;
204  seed_main *= seed_sha256;
205  seed_main *= seed_sha512;
206  if ( seed_secondary )
207  seed_main *= seed_secondary;
208 // puts("Fatto... sto uscendo");
209  // NO recalc anymore
210  password_already_calc = __GIM_YES;
211  char message[256];
212  sprintf( message , "Calculated. main_seed is 0x%ld" , seed_main );
213  gim_error->set( "gim_password_obj::calc" , message );
214  }
215 }

242 void gim_password_obj::prepare_prime( int size ) {
243  int c,j;
244  gim_time_obj t;
245  _gim_handler * hand;
246  char message[256];
247  char tmp[256];
248 
249  if ( ( size <= 2048 ) && ( size >= 256) ) {
250  __GIM_CLEAR( Prime , size , int );
251  if ( env_data != NULL ) {
252  sprintf( tmp , "%s%s" , env_data->general_home , gim_conf->GetKeySTR( "prime" , "filename" ) );
253  if ( gim_file_manager->exist( tmp ) == __GIM_EXIST ) {
254  hand = gim_file_manager->open( tmp , __GIM_FILE_MAPPING , __GIM_READ );
255  memcpy( Prime , hand->fm , sizeof( int ) * size );
256  gim_file_manager->close( hand );
257  }
258  else {
259  gim_error->set( GIM_ERROR_WARNING , "gim_password_obj::prepare_prime" , "Prime data file not found. Regenarating prime array" , __GIM_ERROR );
260  t.start_usec();
261  for( j = 1 , c = 0 ; c < size ; j++ ) {
262  if ( isPrime( j ) )
263  Prime[c++] = j;
264  }
265  sprintf( message, "Prepared %d prime number in %s" , size , t.get_usec() );
266  gim_error->set( "gim_password_obj::prepare_prime" , message );
267  hand = gim_file_manager->open( tmp , __GIM_FILE_POINTER , __GIM_WRITE );
268  fwrite( Prime , sizeof( int ) * size , 1 , hand->fp );
269  gim_file_manager->close( hand );
270  }
271  __GIM_CLEAR( message , 256 , char );
272  for( c = 0 ; c < 20 ; c++ )
273  sprintf( message, "%s%-3d" , message , Prime[c] );
274  strcat( message , "..." );
275  gim_error->set( "gim_password_obj::prepare_prime" , message );
276  }
277  }
