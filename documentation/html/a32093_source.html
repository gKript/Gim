167  inline _gim_Ulong MTRand::randInt() {
168  unsigned long y;
169  static unsigned long mag01[2]={0x0, MATRIX_A};
170  /* mag01[x] = x * MATRIX_A for x=0,1 */
171 
172  if (mti >= MT_N) { /* generate MT_N words at one time */
173  int kk;
174 
175  if (mti == MT_N+1) /* if sgenrand() has not been called, */
176  randInt(4357); /* a default initial seed is used */
177 
178  for (kk=0;kk<MT_N-MT_M;kk++) {
179  y = (mt[kk]&UPPER_MASK)|(mt[kk+1]&LOWER_MASK);
180  mt[kk] = mt[kk+MT_M] ^ (y >> 1) ^ mag01[y & 0x1];
181  }
182  for (;kk<MT_N-1;kk++) {
183  y = (mt[kk]&UPPER_MASK)|(mt[kk+1]&LOWER_MASK);
184  mt[kk] = mt[kk+(MT_M-MT_N)] ^ (y >> 1) ^ mag01[y & 0x1];
185  }
186  y = (mt[MT_N-1]&UPPER_MASK)|(mt[0]&LOWER_MASK);
187  mt[MT_N-1] = mt[MT_M-1] ^ (y >> 1) ^ mag01[y & 0x1];
188 
189  mti = 0;
190  }
191 
192  y = mt[mti++];
193  y ^= TEMPERING_SHIFT_U(y);
194  y ^= TEMPERING_SHIFT_S(y) & TEMPERING_MASK_B;
195  y ^= TEMPERING_SHIFT_T(y) & TEMPERING_MASK_C;
196  y ^= TEMPERING_SHIFT_L(y);
197 
198  return y;
199  }

218  inline void MTRand::seed( const _gim_Ulong oneSeed ) {
219  /* setting initial seeds to mt[MT_N] using */
220  /* the generator Line 25 of Table 1 in */
221  /* [KNUTH 1981, The Art of Computer Programming */
222  /* Vol. 2 (2nd Ed.), pp102] */
223  mt[0]= oneSeed & 0xffffffff;
224  for (mti=1; mti<MT_N; mti++)
225  mt[mti] = (69069 * mt[mti-1]) & 0xffffffff;
226  }

245  inline _gim_Ulong MTRand::hash( time_t t, clock_t c )
246  {
247  // Get a _gim_Ulong from t and c
248  // Better than _gim_Ulong(x) in case x is floating point in [0,1]
249  // Based on code by Lawrence Kirby (fred@genesis.demon.co.uk)
250 
251  static _gim_Ulong differ = 0; // guarantee time-based seeds will change
252 
253  _gim_Ulong h1 = 0;
254  unsigned char *p = (unsigned char *) &t;
255  for( size_t i = 0; i < sizeof(t); ++i )
256  {
257  h1 *= UCHAR_MAX + 2U;
258  h1 += p[i];
259  }
260  _gim_Ulong h2 = 0;
261  p = (unsigned char *) &c;
262  for( size_t j = 0; j < sizeof(c); ++j )
263  {
264  h2 *= UCHAR_MAX + 2U;
265  h2 += p[j];
266  }
267  return ( h1 + differ++ ) ^ h2;
268  }
