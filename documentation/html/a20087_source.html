75 _gim_flag prsr_lexical_class::scan_line ( char * line ) {
76  if ( lex_type == __LEX_A ) {
77  if ( Syntax.Feof == __GIM_ON )
78  return __END;
79  else if ( strlen( line ) == 1 )
80  return __BLANK;
81  char * tmp = strdup( char_filter( line , ' ' ) );
82  if ( is_in_string ( tmp , "#" ) )
83  return __COMMENT;
84  else if ( is_in_string ( tmp , "<" ) )
85  return __TITLE_O ;
86  else if ( is_in_string ( tmp , ">" ) )
87  return __TITLE_C ;
88  else if ( is_in_string ( tmp , "[" ) )
89  return __SECTION_O ;
90  else if ( is_in_string ( tmp , "]" ) )
91  return __SECTION_C ;
92  else if ( is_in_string ( tmp , "=" ) )
93  return __KEY ;
94  else if ( ( ! is_in_string( tmp , "#=[]<>" ) ) && ( strlen( tmp ) > 0 ) )
95  return __SYNTAX_ERROR;
96  return __UNKNOWN;
97  }
98  if ( lex_type == __LEX_B ) {
99  if ( Syntax.Feof == __GIM_ON )
100  return __END;
101  else if ( strlen( line ) == 1 )
102  return __BLANK;
103  char * tmp = strdup( char_filter( line , ' ' ) );
104  if ( is_in_string ( tmp , "<" ) )
105  return __TITLE_O ;
106  else if ( is_in_string ( tmp , "#" ) )
107  return __COMMENT;
108  else if ( is_in_string ( tmp , ">" ) )
109  return __TITLE_C ;
110  else if ( ( is_in_string( tmp , "::" ) ) && ( is_in_string( tmp , "=" ) ) )
111  return __KEY ;
112  else if ( ( ! is_in_string( tmp , "#=:<>" ) ) && ( strlen( tmp ) > 0 ) )
113  return __SYNTAX_ERROR;
114  return __UNKNOWN;
115  }
116  return __UNKNOWN;
117 }

284 _gim_flag prsr_lexical_class::syntax_a( const char * filename ) {
285  unsigned int Line_number = 0;
286  char * line , tmp_error[512];
287  char last_section_open[256];
288  FILE * tmpfp;
289  _gim_flag error = __GIM_OK;
290  tmpfp = fopen( filename , "rb" );
291  if ( ! tmpfp ) return __GIM_NOT_EXIST;
292  Syntax.Title = __GIM_CLOSE;
293  Syntax.Section = __GIM_CLOSE;
294  Syntax.Feof = __GIM_OFF;
295  while ( ( ( Syntax.Feof == __GIM_OFF ) && ( error == 0 ) ) ) {
296  line = strdup( get_line ( tmpfp ) );
297  switch ( scan_line ( line ) ) {
298  case __BLANK : {
299  if ( Syntax.Section == __GIM_OPEN ) {
300  sprintf( tmp_error , "blank line inside a section : %s::%d\n" , filename , Line_number );
301  gim_error->set( "prsr_lexical_class::syntax_a" , tmp_error );
302  }
303  break;
304  }
305  case __TITLE_O : {
306  if ( Syntax.Title == __GIM_OPEN ) {
307  sprintf( tmp_error , "received title twice : %s::%d\n" , filename , Line_number );
308  gim_error->set( GIM_ERROR_CRITICAL , "prsr_lexical_class::syntax_a" , tmp_error , __GIM_ERROR );
309  error = __SYNTAX_ERROR;
310  }
311  Syntax.Title = __GIM_OPEN;
312  break;
313  }
314  case __TITLE_C : {
315  if ( Syntax.Section == __GIM_OPEN ) {
316  sprintf( tmp_error , "received a title close with a section yet opened : %s::%d\n" , filename , Line_number );
317  gim_error->set( GIM_ERROR_CRITICAL , "prsr_lexical_class::syntax_a" , tmp_error , __GIM_ERROR );
318  error = __SYNTAX_ERROR;
319  }
320  Syntax.Title = __GIM_CLOSE;
321  Syntax.Feof = __GIM_ON;
322  break;
323  }
324  case __SECTION_O : {
325  char *tmp;
326  if ( Syntax.Section == __GIM_OPEN ) {
327  sprintf( tmp_error , "Syntax error : last section open : %s\n" , last_section_open );
328  gim_error->set( GIM_ERROR_CRITICAL , "prsr_lexical_class::syntax_a" , tmp_error , __GIM_ERROR );
329  sprintf( tmp_error , "received open section twice : %s::%d\n" , filename , Line_number );
330  gim_error->set( GIM_ERROR_CRITICAL , "prsr_lexical_class::syntax_a" , tmp_error , __GIM_ERROR );
331  error = __SYNTAX_ERROR;
332  }
333  if ( Syntax.Title == __GIM_CLOSE ) {
334  sprintf( tmp_error , "received open section before a title : %s::%d\n" , filename , Line_number );
335  gim_error->set( GIM_ERROR_CRITICAL , "prsr_lexical_class::syntax_a" , tmp_error , __GIM_ERROR );
336  error = __SYNTAX_ERROR;
337  }
338  tmp = strdup( line );
339  tokenizer( tmp , PRSR_LEX_SEPARATORS );
340  if ( strlen( Tok[1] ) == 0 ) {
341  sprintf( tmp_error , "received a section open without a section name : %s::%d\n" , filename , Line_number );
342  gim_error->set( GIM_ERROR_CRITICAL , "prsr_lexical_class::syntax_a" , tmp_error , __GIM_ERROR );
343  error = __SYNTAX_ERROR;
344  }
345  strcpy( last_section_open , Tok[1] );
346  Syntax.Section = __GIM_OPEN;
347  break;
348  }
349  case __SECTION_C : {
350  __GIM_CLEAR( last_section_open , PRSR_MAX_LINE , char );
351  Syntax.Section = __GIM_CLOSE;
352  break;
353  }
354  case __KEY : {
355  char *tmp;
356  if ( Syntax.Section == __GIM_CLOSE ) {
357  sprintf( tmp_error , "received a key with no section open : %s::%d\n" , filename , Line_number );
358  gim_error->set( GIM_ERROR_CRITICAL , "prsr_lexical_class::syntax_a" , tmp_error , __GIM_ERROR );
359  error = __SYNTAX_ERROR;
360  }
361  tmp = strdup( line );
362  tokenizer( tmp , PRSR_LEX_SEPARATORS );
363  if ( strlen( Tok[2] ) || strlen( Tok[3] ) || strlen( Tok[4] ) ) {
364  sprintf( tmp_error , "too much token in line : %s::%d\n" , filename , Line_number );
365  gim_error->set( GIM_ERROR_CRITICAL , "prsr_lexical_class::syntax_a" , tmp_error , __GIM_ERROR );
366  error = __SYNTAX_ERROR;
367  }
368  else {
369  if ( ( ! strlen( Tok[0] ) ) || ( ! strlen( Tok[1] ) ) ) {
370  if ( ! strlen( Tok[0] ) ) {
371  sprintf( tmp_error , "received a value without a key : %s::%d\n" , filename , Line_number );
372  gim_error->set( GIM_ERROR_CRITICAL , "prsr_lexical_class::syntax_a" , tmp_error , __GIM_ERROR );
373  }
374  if ( ! strlen( Tok[1] ) ) {
375  sprintf( tmp_error , "received a key without a value : %s::%d\n" , filename , Line_number );
376  gim_error->set( GIM_ERROR_CRITICAL , "prsr_lexical_class::syntax_a" , tmp_error , __GIM_ERROR );
377  }
378  gim_error->set( GIM_ERROR_CRITICAL , "prsr_lexical_class::syntax_a" , "I suggest to check the file" , __GIM_ERROR );
379  error = __SYNTAX_ERROR;
380  }
381  }
382  break;
383  }
384  case __COMMENT : {
385  break;
386  }
387  case __SYNTAX_ERROR : {
388  sprintf( tmp_error , "unknow kind of line : %s::%d\n" , filename , Line_number );
389  gim_error->set( GIM_ERROR_CRITICAL , "prsr_lexical_class::syntax_a" , tmp_error , __GIM_ERROR );
390  error = __SYNTAX_ERROR;
391  }
392  case __END : {
393  if ( Syntax.Title == __GIM_CLOSE ) {
394  sprintf( tmp_error , "received the end of file before a title : %s::%d\n" , filename , Line_number );
395  gim_error->set( GIM_ERROR_CRITICAL , "prsr_lexical_class::syntax_a" , tmp_error , __GIM_ERROR );
396  error = __SYNTAX_ERROR;
397  }
398  if ( strlen( line ) && ( is_in_string( ']' , line ) ) ) Syntax.Section = __GIM_CLOSE;
399  if ( Syntax.Section == __GIM_OPEN ) {
400  sprintf( tmp_error , "unespectly received the end of file with a section yet opened : %s::%d\n" , filename , Line_number );
401  gim_error->set( GIM_ERROR_CRITICAL , "prsr_lexical_class::syntax_a" , tmp_error , __GIM_ERROR );
402  error = __SYNTAX_ERROR;
403  }
404  gim_error->set( GIM_ERROR_WARNING , "prsr_lexical_class::syntax_a" , "Warning : maybe the file not contain the title close" , __GIM_ERROR );
405  break;
406  }
407  }
408  Line_number++;
409  }
410  fclose(tmpfp);
411  return error;
412 }
