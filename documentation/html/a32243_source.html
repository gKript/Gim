41 void gim_signal_received( int sig) {
42  if ( fatal_error_in_progress == __GIM_YES )
43  raise( sig );
44  fatal_error_in_progress = __GIM_YES;
45  gim_error->set( GIM_ERROR_CRITICAL , "::gim_signal_received" , "Signal received" , sig );
46  char fault[512] , buf[512];
47  gim_error->set( GIM_ON_FATAL_EXIT , __GIM_YES );
48  strcpy( fault , strsignal( sig ) );
49  snprintf( buf , sizeof( buf ), "Gim : [ %s ]", fault );
50  signal(SIGINT , SIG_DFL );
51  signal(SIGQUIT, SIG_DFL );
52  signal(SIGKILL, SIG_DFL );
53  signal(SIGABRT, SIG_DFL );
54  signal(SIGTERM, SIG_DFL );
55  signal(SIGSEGV, SIG_DFL );
56  gim_error->set( GIM_ERROR_FATAL , "::gim_signal_received" , buf , sig );
57 }

59 void gim_signal_handler( void ) {
60  signal(SIGINT , gim_signal_received );

62  signal(SIGKILL, gim_signal_received );

64  signal(SIGTERM, gim_signal_received );
