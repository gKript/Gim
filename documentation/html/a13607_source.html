81 void gim_time_obj::tv_removal( void ) {
82  tvdiff.tv_sec = tvstop.tv_sec - tvstart.tv_sec;
83  tvdiff.tv_usec = tvstop.tv_usec - tvstart.tv_usec;
84  if (tvdiff.tv_sec < 0 && tvdiff.tv_usec > 0)
85  tvdiff.tv_sec++, tvdiff.tv_usec -= 1000000;
86  if (tvdiff.tv_sec > 0 && tvdiff.tv_usec < 0)
87  tvdiff.tv_sec--, tvdiff.tv_usec += 1000000;
88 }
