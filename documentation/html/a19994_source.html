214 void gim_obj::mt_test( void ) {
215  gim_error->set( "gim_enhanced_base_obj::mt_test" , "MersenneTwister random number generation test..." );
216  gim_time_obj t;
217  _gim_rand mt;
218  int iters = gim_conf->GetKeyINT( "mersenne" , "iterations" );
219  int test[iters];
220  int c = 0;
221  char res[128];
222  time_t sys = ::time( NULL );
223  mt.seed( sys );
224  t.start_usec();
225  for( c = 0 ; c < iters ; c++ ) {
226  test[c] = mt.randInt( 50 );
227  }
228  strcpy( res , t.get_usec() );
229  char message[256];
230  __GIM_CLEAR( message , 256 , char );
231  for( c = 0 ; c < 20 ; c++ )
232  sprintf( message , "%s%2d " ,message , test[c] );
233  strcat( message , "..." );
234  gim_error->set( "gim_enhanced_base_obj::mt_test" , message );
235  sprintf( message , "%d number generated in sec [%s]. Test done" , iters , res );

240 void gim_obj::mt_test_check( void ) {
241  volatile int changed = __GIM_NO;
242  if ( gim_conf->ExistSection( "mersenne" ) == __GIM_EXIST ) {
243  if ( gim_conf->ExistKey( "mersenne" , "test" ) == __GIM_EXIST ) {
244  if ( gim_conf->ExistKey( "mersenne" , "iterations" ) == __GIM_EXIST ) {
245  if ( gim_conf->GetKeyFLAG( "mersenne" , "test" ) == __GIM_YES ) {
246  mt_test();
247  }
248  }
249  }
250  else {
251  gim_conf->AddKeyFlag( "mersenne" , "test" , __GIM_NO );
252  gim_error->set( GIM_ERROR_WARNING , "gim_enhanced_base_obj::mt_test_check" , "Mersenne::test key not found. Test setted NO per default" , __GIM_ERROR );
253  changed = __GIM_YES;
254  }
255  if ( gim_conf->ExistKey( "mersenne" , "iterations" ) == __GIM_NOT_EXIST ) {
256  gim_conf->AddKey( "mersenne" , "iterations" , 200 );
257  gim_error->set( GIM_ERROR_WARNING , "gim_enhanced_base_obj::mt_test_check" , "Mersenne::iterations key not found. Iterations setted 200 per default" , __GIM_ERROR );
258  changed = __GIM_YES;
259  }
260  }
261  else {
262  gim_conf->AddSection( "mersenne" );
263  gim_conf->AddKeyFlag( "mersenne" , "test" , __GIM_NO );
264  gim_error->set( GIM_ERROR_WARNING , "gim_enhanced_base_obj::mt_test_check" , "Mersenne Section not found. Test setted NO per default" , __GIM_ERROR );
265  changed = __GIM_YES;
266  }
267  if ( changed == __GIM_YES ) {
268  gim_home->add_default_comment();
269  gim_conf->Write();
270  }
