133  inline gim_db_obj() {
134  if ( strlen( gim_application_name ) != 0 ) {
135  gim_error->set( "gim_db_obj::gim_db_obj" , "DB : Constructor start" );
136  db = (_gim_db_main *)gim_memory->Alloc( sizeof( _gim_db_main ) , __GIM_MEM_DB_MAIN , __GIM_HIDE );
137  strcpy( name , "" );
138  strcpy( comment , "" );
139  strcpy( file_name , "" );
140  strcpy( file_name_long , "" );
141  db->mode = GIM_DB_PERFORMANCE;
142  db->type = GIM_DB_VOLATILE;
143  db->tables_number = 0;
144  strcpy( db->pin_table , "" );
145  db->pin = __GIM_NO;
146  db->conf = new _gim_prsr;
147  db->tables = new _gim_list;
148  cs = new gim_checksum_obj;
149  lex = new _gim_lex;
150  initiated = __GIM_NO;
151  changed = __GIM_NO;
152  syntax = (SyntaxCoerence_obj *)gim_memory->Alloc( sizeof( SyntaxCoerence_obj ) , __GIM_MEM_DB_MAIN , __GIM_HIDE );
153  syntax->Title = __GIM_OFF;
154  syntax->Section = __GIM_OFF;
155  syntax->Feof = __GIM_OFF;
156  gdbs_script = new _gim_list;
157  gdbs_run = __GIM_NOT_RUNNING;
158  gim_error->set( "gim_db_obj::gim_db_obj" , "DB : Constructor end" );
159  }
160  else
161  gim_error->set( GIM_ERROR_FATAL , "gim_db_obj::gim_db_obj" , "DB : initialization error : use of db is not allowed without the application name" , __GIM_ERROR );
162  };

166  inline ~gim_db_obj() {
167  delete lex;
168  delete cs;
169  if ( strlen( name ) ) {
170  gim_error->Set( GIM_ERROR_MESSAGE , "gim_db_obj::~gim_db_obj" , "DB [%s] : start to shutting down..." , name );
171  }
172  else
173  gim_error->set( "gim_db_obj::~gim_db_obj" , "DB : start to shutting down..." );
174  if ( ( db->type == GIM_DB_PERMANENT ) && ( changed == __GIM_YES ) ) {
175  gim_error->set( "gim_db_obj::~gim_db_obj" , "DB : Something is changed in configuration. Updating" );
176  db->conf->Write();
177  }
178  if ( db->tables->items() ) {
179  gim_error->set( "gim_db_obj::~gim_db_obj" , "DB : Some allocated Table found. Deallocating" );
180  db->tables->rewind();
181  for ( _gim_int32 c = 0 ; c < db->tables_number ; c++ ) {
182  _gim_db_table * Ttmp = (_gim_db_table *)db->tables->get_item();
183  Ttmp->Tstruct->Down();
184  Ttmp->Tdata->Down();
185 /* if ( Ttmp->fields->items() )
186  Ttmp->fields->destroy_list();
187 */ delete Ttmp->fields;
188  if ( Ttmp->records->items() ) {
189  gim_error->Set( GIM_ERROR_MESSAGE , "gim_db_obj::~gim_db_obj" , "DB : Found some records [%d]. Deallocating..." , Ttmp->records->items() );
190  for( _gim_int32 r = 1 ; r <= Ttmp->records->items() ; r++ ) {
191  _gim_db_record * tr = (_gim_db_record *)Ttmp->records->get_item();
192  tr->values->destroy_list();
193  Ttmp->records->next_item();
194  }
195  Ttmp->records->destroy_list();
196  }
197  delete Ttmp->records;
198  gim_memory->Free( Ttmp );
199  db->tables->next_item();
200  }
201  }
202  delete gdbs_script;
203  gim_memory->Free ( syntax );
204  delete db->tables;
205  db->conf->Down();
206  delete db->conf;
207  gim_memory->Free( db );
208  gim_error->set( "gim_db_obj::~gim_db_obj" , "DB : shutted down" );
209  };
210  };
