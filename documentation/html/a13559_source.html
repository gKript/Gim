145 _gim_flag gim_html_obj::new_page( const char * title , const char * comment , const char * file_name ) {
146  gim_error->set( "gim_html_obj::new_page" , "New html page requested..." );
147  strcpy( page->title , title );
148  strcpy( page->comment , comment );
149  strcpy( page->file_name , file_name );
150  gim_error->set( "gim_html_obj::new_page" , "Parameter setted" );
151  page->fh = gim_file_manager->open( page->file_name , __GIM_FILE_POINTER , __GIM_WRITE );
152  if ( page->fh == NULL ) {
153  gim_error->set( GIM_ERROR_CRITICAL , "gim_html_obj::new_page" , "File NOT CORRECTLY opened for writing" , __GIM_ERROR );
154  return __GIM_ERROR;
155  }
156  else {
157  gim_error->set( "gim_html_obj::new_page" , "File correctly opened for writing... Now i define the initial html code" );
158  Export_tag_open( "html" );
159  Export_tag_open( "head" );
160  Export_with_tag( "title" , page->title );
161  Export_tag_close( "head" );
162  gim_error->set( "gim_html_obj::new_page" , "done!!!" );
163  }
164  return __GIM_OK;
165 }

168 _gim_flag gim_html_obj::close_page( void ) {

177  Export_tag_close( "html" );
178  gim_file_manager->close( page->fh );
