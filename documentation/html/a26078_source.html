701 _gim_int8 gim_db_obj::gdbs_tokenizer( char * line ) {
702  if ( !strlen( line ) )
703  return __SKIPPED;
704 
705  static unsigned int i;
706  unsigned int a , b = 0;
707  volatile _gim_flag for_exit = __GIM_NO , termination = __GIM_NO , comment = __GIM_NO;
708 
709  i = 0;
710  __GIM_VCLEAR( Tok , 1 , Tok , '\0' );
711  strcat( line , " " );
712  if ( lex->is_in_string( GDBS_TERMINATOR_CHAR , line ) == __GIM_YES )
713  termination = __GIM_YES;
714  for( a = 0 ; ( ( a < ( strlen( line ) - 1 ) ) && ( for_exit == __GIM_NO ) ) ; a++ ) {
715  if ( ( i > 19 ) || ( line[a] == GDBS_STOP_CHAR ) ) {
716  comment = __GIM_YES;
717  break;
718  }
719  if ( lex->is_in_string( line[a] , GDBS_SKIP_STR ) == __GIM_NO ) {
720  if ( lex->is_in_string( line[a] , GDBS_SEPARATOR_STR ) ) {
721  if ( b ) {
722  Tok[i++][b] = '\0' ;
723  b=0;
724  }
725  else
726  continue;
727  }
728  else
729  Tok[i][b++] = line[a];
730  }
731  }
732 
733  gim_error->Set( GIM_ERROR_MESSAGE , "prsr_lexical_class::gdbs_tokenizer" , "Summary: [line: %3d] [Len: %3d] [NoT: %2d] [termination: %d] [comment: %d]" , gdbs_line_num , strlen( line ) , i , termination , comment );
734  if ( i > 19 ) {
735  gim_error->set( GIM_ERROR_CRITICAL , "prsr_lexical_class::gdbs_tokenizer" , "too much token in line" , __GIM_ERROR );
736  return __TOO_MUCH_TOKEN;
737  }
738  else if ( ( ( i <= 1 ) && ( comment == __GIM_NO ) && ( strlen( line ) > 2 ) ) ) {
739  return __SYNTAX_ERROR;
740  }
741  else if ( ( i ) && ( strlen( Tok[0] ) ) && ( termination == __GIM_YES ) ) {
742  gim_error->Set( GIM_ERROR_MESSAGE , "prsr_lexical_class::gdbs_tokenizer" , "Correct GDBS line analyzed [extracted %d token]" , i );
743  return i;
744  }
745  else if ( ( i ) && ( !strlen( Tok[0] ) ) && ( termination == __GIM_YES ) ) {
746  gim_error->set( GIM_ERROR_CRITICAL , "prsr_lexical_class::gdbs_tokenizer" , "Syntax error in GDBS file" , __GIM_ERROR );
747  return __SYNTAX_ERROR;
748  }
749  else if ( ( i ) && ( strlen( Tok[0] ) ) && ( termination == __GIM_NO ) ) {
750  gim_error->set( GIM_ERROR_CRITICAL , "prsr_lexical_class::gdbs_tokenizer" , "Missing line terminator in GDBS file" , __GIM_ERROR );
751  return __MISSING_STOP_CHAR;
752  }
753  return __SKIPPED;
754 }
