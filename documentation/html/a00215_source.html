52  temp_opt->option = c;
53  if ( optarg == NULL )
54  temp_opt->argument = NULL;

61  temp_opt->status = __GIM_OPT_OK;

131 _gim_flag gim_getopt_obj::search( char opt ) {
132  optlist->backup();
133  optlist->rewind();
134  for( ; optlist->get_item() != NULL ; optlist->next_item() ) {
135  _gim_option * tmp = (_gim_option *)optlist->get_item();
136  if ( tmp->option == opt ) {
137  optlist->restore();
138  return __GIM_YES;
139  }
140  }
141  return __GIM_NO;
142 }
