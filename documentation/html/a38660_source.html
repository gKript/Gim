123 _gim_processor * gim_identity_obj::cpu( void ) {
124  FILE * cpuinfo;
125  char buffer[128];
126  char *tmp[5];
127  int prc_n , cur_prc; // temp var for smp system
128  _gim_processor * proc;
129  _gim_processor * tmp_proc;
130  _gim_processor * start_proc;
131 
132  cpuinfo = fopen( "/proc/cpuinfo" , "rb" );
133  if ( cpuinfo == NULL )
134  return NULL;
135  gim_error->set( "gim_identity_obj::cpu" , "Getting processor details..." );
136  proc = (_gim_processor *)gim_memory->Alloc( sizeof( _gim_processor ) , __GIM_MEM_GETID_CPU , __GIM_HIDE);
137  start_proc = proc;
138  start_proc->next = NULL;
139  prc_n = cur_prc = 0;
140  proc->proc_number = 0;
141  strcpy( proc->model_name , "" );
142  while ( fgets( buffer , 128 , cpuinfo ) ) {
143  if ( strlen( buffer ) ) {
144  gim_memory->caller = __GIM_MEM_READ_KEY;
145  Lexical.tokenizer( buffer , ":\n" );
146  if ( strlen( Tok[1] ) ) {
147  tmp[0] = strdup( Tok[0] );
148  Lexical.strend( tmp[0] , '\t' );
149  }
150  if ( strlen( Tok[1] ) )
151  tmp[1] = strdup( Lexical.str_offset_pre( Tok[1] , 1 ) );
152  get_int( "processor" , prc_n );
153  if ( prc_n != cur_prc ) {
154  cur_prc = prc_n;
155  proc->next = (_gim_processor *)gim_memory->Alloc( sizeof( _gim_processor ) , __GIM_MEM_GETID_CPU , __GIM_HIDE);
156  proc = proc->next;
157  proc->proc_number = cur_prc;
158  n_processors = cur_prc + 1;
159  proc->next = NULL;
160  }
161  get_str( "model name" , proc->model_name );
162  get_str( "Processor" , proc->model_name );
163  get_str( "vendor_id" , proc->vendor_id);
164  get_int( "cache size" , proc->cache_size);
165  get_float( "cpu MHz" , proc->cpu_mhz);
166  get_float( "bogomips" , proc->bogomips);
167  get_float( "BogoMIPS" , proc->bogomips);
168  get_int( "model" , proc->model);
169  get_int( "cpu family" , proc->family);
170  get_int( "stepping" , proc->stepping);
171  }
172  if ( proc->cpu_mhz == 0 ) {
173  if ( Lexical.is_in_string( proc->model_name , "arm" ) ) {
174  proc->cpu_mhz = proc->bogomips;
175  proc->bogomips = 0.00;
176  }
177  else {
178  proc->est_cpu_mhz = proc->bogomips / 2.081472727 ;
179  proc->cpu_mhz = proc->est_cpu_mhz;
180  }
181  }
182  cpu_strfamily( proc );
183  }
184  fclose(cpuinfo);
185  return start_proc;
186 }

302 _gim_memory * gim_identity_obj::memory(void) {
303  FILE * procmem;
304  char buffer[128];
305  char tmp[5][256];
306 
307  procmem = fopen( "/proc/meminfo" , "rb" );
308  if ( procmem == NULL )
309  return NULL;
310  gim_error->set( "gim_identity_obj::memory" , "Getting memory details..." );
311  mi = (_gim_memory *)gim_memory->Alloc( sizeof( _gim_memory ) , __GIM_MEM_GETID_MEM , __GIM_HIDE );
312  while ( fgets( buffer , 128 , procmem ) ) {
313  gim_memory->caller = __GIM_MEM_READ_KEY;
314  Lexical.tokenizer( buffer , ":kB\n" );
315  strcpy( tmp[0] , Lexical.char_filter( Tok[0] , ' ' ) );
316  strcpy( tmp[1] , Lexical.char_filter( Tok[1] , ' ' ) );
317  get_int("MemTotal", mi->total);
318  get_int("MemFree", mi->free);
319  get_int("Cached", mi->cached);
320  }
321  fclose(procmem);
322 
323  mi->free += mi->cached;
324  mi->used = mi->total - mi->free ;
325 
326  mi->total *= 1024;
327  mi->cached *= 1024;
328  mi->used *= 1024;
329  mi->free *= 1024;
330 
331  mi->ratio = 100.0 * ( 1.0 - (float)((float)mi->used / (float)mi->total) );
332 
333  return mi;
334 }
