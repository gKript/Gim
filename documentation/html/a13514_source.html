103  gim_memory->Free( currentlist->hand->loadp );
104  gim_error->set( "gim_file_manager_obj::release_item_list" , "Unloaded succesfully" );

108  }

112  }

114  gim_error->set( "gim_file_manager_obj::release_item_list" , message );

262  del_item( item );
263  //~ gim_error->set( "gim_file_manager_obj::close" , "Item memory releasing" );

265 
266  }
267  else
268  gim_error->set( GIM_ERROR_CRITICAL , "gim_file_manager_obj::close" , "item not found. Maybe never opened or close twice" , __GIM_ERROR );

324 _gim_handler * gim_file_manager_obj::open( const char * filename , _gim_flag method , _gim_flag mode ) {
325  gim_file_manager_list * item;
326  char message[256];
327  if ( exist( filename ) == __GIM_NO ) {
328  gim_error->set( GIM_ERROR_CRITICAL , "gim_file_manager_obj::open" , "File not found" , __GIM_NOT_EXIST );
329  return NULL;
330  }
331  if ( mode != __GIM_WRITE ) {
332  if ( gim_utils.isregular( filename ) == __GIM_NO ) {
333  gim_error->set( GIM_ERROR_CRITICAL , "gim_file_manager_obj::open" , "This file is not a regular file" , __GIM_NOT_EXIST );
334  return NULL;
335  }
336  }
337  sprintf( message , "Opening file [%s] in %s %s..." , Lexical.string_trunc ( filename , 30 ) , method_str( method ) , mode_str( mode ) );
338  gim_error->set( "gim_file_manager_obj::open" , message );
339  item = new_item();
340  item->hand = (_gim_handler *)gim_memory->Alloc( sizeof( _gim_handler ) , __GIM_MEM_FMAN_OPEN , __GIM_HIDE );
341  strcpy( item->hand->name , filename );
342  if ( mode == __GIM_READ ) {
343  stat( item->hand->name , &(item->st) );
344  item->hand->size = item->st.st_size;
345  }
346  item->mode = mode;
347  item->method = method;
348  switch( method ) {
349  case __GIM_FILE_DESCRIPTOR : {
350  open_fd( item , mode );
351  if ( item->hand->fd == 0 ) {
352  gim_error->set( GIM_ERROR_CRITICAL , "gim_file_manager_obj::open" , "Some problem opening file" , __GIM_NOT_OK );
353  del_item( item );
354  return NULL;
355  }
356  break;
357  }
358  case __GIM_FILE_POINTER : {
359  open_fp( item , mode );
360  if ( item->hand->fp == NULL ) {
361  gim_error->set( GIM_ERROR_CRITICAL , "gim_file_manager_obj::open" , "Some problem opening file" , __GIM_NOT_OK );
362  del_item( item );
363  return NULL;
364  }
365  break;
366  }
367  case __GIM_FILE_MAPPING : {
368  if ( mode == __GIM_READ ) {
369  open_fm( item );
370  if ( item->hand->fm == NULL ) {
371  gim_error->set( GIM_ERROR_CRITICAL , "gim_file_manager_obj::open" , "Some problem opening file" , __GIM_NOT_OK );
372  del_item( item );
373  return NULL;
374  }
375  }
376  else {
377  gim_error->set( GIM_ERROR_CRITICAL , "gim_file_manager_obj::open" , "In file mapping method is possible to open a file in __GIM_READ only" , __GIM_NOT_OK );
378  return NULL;
379  }
380  break;
381  }
382  default : {
383  gim_error->set( GIM_ERROR_CRITICAL , "gim_file_manager_obj::open" , "Incorrect method" , __GIM_NOT_OK );
384  return NULL;
385  }
386  }
387  return item->hand;
388 }

470 _gim_buffer gim_file_manager_obj::load( _gim_handler * hand ) {
471  gim_file_manager_list * tmp = NULL;
472  tmp = item( hand->name );
473  if ( tmp != NULL ) {
474  if ( tmp->hand->loadp != NULL ) {
475  if ( gim_memory->Check( hand->loadp ) == __GIM_EXIST ) {
476  char message[256];
477  sprintf( message , "File already loaded [0x%p]" , hand->loadp );
478  gim_error->set( GIM_ERROR_WARNING , "gim_file_manager_obj::load" , message , __GIM_NOT_OK );
479  return hand->loadp;
480  }
481  else
482  hand->loadp = NULL;
483  }
484  if ( tmp->mode == __GIM_READ ) {
485  switch ( tmp->method ) {
486  case __GIM_FILE_DESCRIPTOR : {
487  hand->loadp = (char *)gim_memory->Alloc( hand->size * sizeof( char ) , __GIM_MEM_FMAN_LOAD , __GIM_HIDE );
488  read( hand->fd , hand->loadp , hand->size );
489  break;
490  }
491  case __GIM_FILE_POINTER : {
492  hand->loadp = (char *)gim_memory->Alloc( hand->size * sizeof( char ) , __GIM_MEM_FMAN_LOAD , __GIM_HIDE );
493  unsigned int tot = fread( hand->loadp , 1 , hand->size , hand->fp );
494  if ( tot != hand->size ) {
495  char message[256];
496  sprintf( message , "Some problem with fread. Read %d/%d bytes" , tot , hand->size );
497  gim_error->set( GIM_ERROR_CRITICAL , "gim_file_manager_obj::load" , message , __GIM_ERROR );
498  }
499  break;
500  }
501  case __GIM_FILE_MAPPING : {
502  hand->loadp = (char *)gim_memory->Alloc( hand->size * sizeof( char ) , __GIM_MEM_FMAN_LOAD , __GIM_HIDE );
503  memcpy( hand->loadp , hand->fm , hand->size );
504  break;
505  }
506  }
507  gim_error->set( "gim_file_manager_obj::load" , "Loaded succesfully" );
508  }
509  else {
510  if ( tmp->mode != __GIM_READ ) {
511  gim_error->set( "gim_file_manager_obj::load" , "Load is possible in __GIM_READ only" );
512  }
513  }
514  return hand->loadp;
515  }
516  return NULL;
517 }

521 _gim_flag gim_file_manager_obj::flush( _gim_handler * hand , _gim_buffer data , int data_len ) {
522  if ( data == NULL ) {
523  gim_error->set( GIM_ERROR_CRITICAL , "gim_file_obj::flush" , "The source buffer cannot be NULL" , __GIM_ERROR );
524  return __GIM_NOT_OK;
525  }
526  gim_file_manager_list * tmp = NULL;
527  tmp = item( hand->name );
528  if ( tmp->mode == __GIM_WRITE ) {
529  switch ( tmp->method ) {
530  case __GIM_FILE_DESCRIPTOR : {
531  write( hand->fd , data , data_len );
532  break;
533  }
534  case __GIM_FILE_POINTER : {
535  fwrite( data , data_len , 1 , hand->fp );
536  break;
537  }
538  case __GIM_FILE_MAPPING : {
539  gim_error->set( GIM_ERROR_WARNING , "gim_file_obj::flush" , "Flush is not possible in __GIM_FILE_MAPPING" , __GIM_ERROR );
540  return __GIM_NOT_OK;
541  }
542  }
543  gim_error->set( "gim_file_obj::flush" , "Flushed succesfully" );
544  return __GIM_OK;
545  }
546  else
547  if ( tmp->mode != __GIM_WRITE ) {
548  gim_error->set( GIM_ERROR_WARNING , "gim_file_obj::flush" , "Flush is possible in __GIM_WRITE only" , __GIM_ERROR );

703 _gim_buffer gim_file_obj::load( void ) {
704  if ( handler->loadp != NULL ) {
705  gim_error->set( GIM_ERROR_WARNING , "gim_file_obj::load" , "File already loaded" , __GIM_ERROR );
706  return handler->loadp;
707  }
708  if ( ( mode == __GIM_READ ) && ( exist == __GIM_EXIST ) ) {
709  switch ( method ) {
710  case __GIM_FILE_DESCRIPTOR : {
711  handler->loadp = (char *)gim_memory->Alloc( handler->size * sizeof( char ) , __GIM_MEM_FOBJ_LOAD , __GIM_HIDE );
712  if ( handler->loadp == NULL ) return NULL;
713  read( handler->fd , handler->loadp , handler->size );
714  break;
715  }
716  case __GIM_FILE_POINTER : {
717  handler->loadp = (char *)gim_memory->Alloc( handler->size * sizeof( char ) , __GIM_MEM_FOBJ_LOAD , __GIM_HIDE );
718  if ( handler->loadp == NULL ) return NULL;
719  fread( handler->loadp , handler->size , 1 , handler->fp );
720  break;
721  }
722  case __GIM_FILE_MAPPING : {
723  handler->loadp = (char *)gim_memory->Alloc( handler->size * sizeof( char ) , __GIM_MEM_FOBJ_LOAD , __GIM_HIDE );
724  if ( handler->loadp == NULL ) return NULL;
725  memcpy( handler->loadp , handler->fm , handler->size );
726  break;
727  }
728  }
729  gim_error->set( "gim_file_obj::load" , "Loaded succesfully" );
730  }
731  else {
732  if ( mode != __GIM_READ ) {
733  gim_error->set( GIM_ERROR_WARNING , "gim_file_obj::load" , "Load is possible in __GIM_READ only" , __GIM_ERROR );
734  }

736  gim_error->set( GIM_ERROR_WARNING , "gim_file_obj::load" , "The file not exist" , __GIM_ERROR );
