151 _gim_flag gim_ascii_file_obj::seek( _gim_int32 position ) {
152  _gim_int32 c = 0;
153  if ( position > this->lenght() ) {
154  gim_error->set( GIM_ERROR_WARNING , "gim_ascii_file_obj::seek" , "Requested position is beyond the end of buffer" , __GIM_ERROR );
155  return __GIM_ERROR;
156  }
157  if ( ( dimension > 0 ) && ( chrbuf != NULL ) ) {
158  while ( ( chrbuf[c] != '\0' ) || ( c == position ) ) {
159  index = c++;
160  }
161  if ( c == position )
162  return __GIM_OK;
163  else {
164  gim_error->set( GIM_ERROR_WARNING , "gim_ascii_file_obj::seek" , "Reached the end of buffer before the requested position" , __GIM_ERROR );
