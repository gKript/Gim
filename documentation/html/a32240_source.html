147 static void convert_to_bigendian( void* data, int len ) {
148 /* test endianness */
149  word32 test_value = 0x01;
150  byte* test_as_bytes = (byte*) &test_value;
151  int little_endian = test_as_bytes[ 0 ];
152  word32 temp;
153  byte* temp_as_bytes = (byte*) &temp;
154  word32* data_as_words = (word32*) data;
155  byte* data_as_bytes;
156  int i;
157  if ( little_endian ) {
158  len /= 4;
159  for ( i = 0; i < len; i++ ) {
160  temp = data_as_words[ i ];
161  data_as_bytes = (byte*) &( data_as_words[ i ] );
162  data_as_bytes[ 0 ] = temp_as_bytes[ 3 ];
163  data_as_bytes[ 1 ] = temp_as_bytes[ 2 ];
164  data_as_bytes[ 2 ] = temp_as_bytes[ 1 ];
165  data_as_bytes[ 3 ] = temp_as_bytes[ 0 ];
166  }
167  }

169  automatically does the right thing for SHA1 */

179 static void SHA512_transform( SHA512_ctx* ctx ) {
180  int t;
181  word32 Ah = ctx->H[ 0 ];
182  word32 Al = ctx->H[ 1 ];
183  word32 Bh = ctx->H[ 2 ];
184  word32 Bl = ctx->H[ 3 ];
185  word32 Ch = ctx->H[ 4 ];
186  word32 Cl = ctx->H[ 5 ];
187  word32 Dh = ctx->H[ 6 ];
188  word32 Dl = ctx->H[ 7 ];
189  word32 Eh = ctx->H[ 8 ];
190  word32 El = ctx->H[ 9 ];
191  word32 Fh = ctx->H[ 10 ];
192  word32 Fl = ctx->H[ 11 ];
193  word32 Gh = ctx->H[ 12 ];
194  word32 Gl = ctx->H[ 13 ];
195  word32 Hh = ctx->H[ 14 ];
196  word32 Hl = ctx->H[ 15 ];
197  word32 T1h, T1l, T2h, T2l;
198  word32 W[ 160 ]; /* Stored as hi, lo */
199  memcpy( W, ctx->M, 32*sizeof(word32) );
200  for ( t = 16; t < 80; t++ ) {
201 // W[ t ] = sig1(W[t-2]) + W[t-7] + sig0(W[t-15]) + W[t-16];
202  T1h = sig1h(W[2*t-4],W[2*t-3]); T1l = sig1l(W[2*t-4],W[2*t-3]);
203  ADDC( T1h, T1l, W[2*t-14], W[2*t-13] );
204  ADDC( T1h, T1l, sig0h(W[2*t-30],W[2*t-29]), sig0l(W[2*t-30],W[2*t-29]) );
205  ADDC( T1h, T1l, W[2*t-32], W[2*t-31] );
206  W[2*t] = T1h; W[2*t+1] = T1l;
207  }
208  for ( t = 0; t < 80; t++ ) {
209 // T1 = H + SIG1(E) + Ch(E,F,G) + K[t] + W[t];
210  T1h = Hh; T1l = Hl;
211  ADDC( T1h, T1l, SIG1h(Eh,El), SIG1l(Eh,El) );
212  ADDC( T1h, T1l, Choice(Eh,Fh,Gh), Choice(El,Fl,Gl));
213  ADDC( T1h, T1l, K[2*t], K[2*t+1] );
214  ADDC( T1h, T1l, W[2*t], W[2*t+1] );
215 // T2 = SIG0(A) + Maj(A,B,C);
216  T2h = SIG0h(Ah,Al); T2l = SIG0l(Ah,Al);
217  ADDC( T2h, T2l, Maj(Ah,Bh,Ch), Maj(Al,Bl,Cl) );
218  Hh = Gh; Hl = Gl;
219  Gh = Fh; Gl = Fl;
220  Fh = Eh; Fl = El;
221 // E = D + T1;
222  Eh = Dh; El = Dl;
223  ADDC( Eh, El, T1h, T1l );
224  Dh = Ch; Dl = Cl;
225  Ch = Bh; Cl = Bl;
226  Bh = Ah; Bl = Al;
227 // A = T1 + T2;
228  Ah = T1h; Al = T1l;
229  ADDC( Ah, Al, T2h, T2l );
230  }
231  ADDC( ctx->H[ 0], ctx->H[ 1], Ah, Al );
232  ADDC( ctx->H[ 2], ctx->H[ 3], Bh, Bl );
233  ADDC( ctx->H[ 4], ctx->H[ 5], Ch, Cl );
234  ADDC( ctx->H[ 6], ctx->H[ 7], Dh, Dl );
235  ADDC( ctx->H[ 8], ctx->H[ 9], Eh, El );
236  ADDC( ctx->H[10], ctx->H[11], Fh, Fl );
237  ADDC( ctx->H[12], ctx->H[13], Gh, Gl );
238  ADDC( ctx->H[14], ctx->H[15], Hh, Hl );
239 }

241 void gim_sha512_obj::SHA512_update( SHA512_ctx* ctx, const void* vdata, word32 data_len ) {
242  const byte* data = ( const byte * ) vdata;
243  word32 use;
244  word32 low_bits;
245 /* convert data_len to bits and add to the 128 bit word formed by lbits
246  and hbits */
247  ctx->hbits += data_len >> 29;
248  low_bits = data_len << 3;
249  ctx->lbits += low_bits;
250  if ( ctx->lbits < low_bits ) { ctx->hbits++; }
251 /* deal with first block */
252  use = min( 128 - ctx->mlen, data_len );
253  memcpy( ctx->M + ctx->mlen, data, use );
254  ctx->mlen += use;
255  data_len -= use;
256  data += use;
257  while ( ctx->mlen == 128 ) {
258  convert_to_bigendian( (word32*)ctx->M, 128 );
259  SHA512_transform( ctx );
260  use = min( 128, data_len );
261  memcpy( ctx->M, data, use );
262  ctx->mlen = use;
263  data_len -= use;
264  data += use; /* was missing */
265  }
266 }

268 void gim_sha512_obj::SHA512_final( SHA512_ctx* ctx ) {
269  if ( ctx->mlen < 128-16 ) {
270  ctx->M[ ctx->mlen ] = 0x80; ctx->mlen++;
271  memset( ctx->M + ctx->mlen, 0x00, 128-8 - ctx->mlen );
272  convert_to_bigendian( ctx->M, 128-8 );
273  }
274  else {
275  ctx->M[ ctx->mlen ] = 0x80;
276  ctx->mlen++;
277  memset( ctx->M + ctx->mlen, 0x00, 128 - ctx->mlen );
278  convert_to_bigendian( ctx->M, 128 );
279  SHA512_transform( ctx );
280  memset( ctx->M, 0x00, 128-8 );
281  }
282  memcpy( ctx->M + 128-8, (void*)(&(ctx->hbits)), 8 );
283  SHA512_transform( ctx );
284 }
