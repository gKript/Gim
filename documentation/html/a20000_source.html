49 char * gim_checksum_obj::md5( char * buffer , _gim_Uint32 size ) {
50  char digest[16];
51  static char hex[33];
52  MD5_CTX c;
53  _gim_Uint8 i;
54  gim_md5_obj md5;
55  md5.MD5Init( &c );
56  md5.MD5Update( &c , ( unsigned char * ) buffer , size );
57  md5.MD5Final( (unsigned char *) digest , &c );
58  for( i = 0 ; i < 16 ; i++ )
59  snprintf( ( hex + ( 2 * i ) ) , 3 , "%02x" , ( unsigned char ) digest[i] );
60  return hex;
