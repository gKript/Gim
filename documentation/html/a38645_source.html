96 _gim_Uint32 gim_gkdp_obj::make_gkdp( const char * dir_name ) {
97  struct dirent ** eps;
98  int n , cnt;
99  char TmpFileName[1024];
100  n = scandir ( dir_name , &eps , NULL , alphasort );
101  if ( n ) {
102  char message[256];
103  currentlist = add();
104  strcpy( currentlist->name , dir_name );
105  currentlist->size = 0;
106  sprintf( message , "Adding root %s" , dir_name );
107  gim_error->set( "gim_gkdp_obj::make_gkdp" , message );
108  for ( cnt = 0 ; cnt < n ; cnt++ ) {
109  struct stat tmp;
110  sprintf( TmpFileName , "%s%s" , dir_name , eps[cnt]->d_name );
111  if ( ( ! strcmp( "." , eps[cnt]->d_name ) ) || ( ! strcmp( ".." , eps[cnt]->d_name ) ) )
112  continue;
113  if ( stat( TmpFileName , &tmp ) )
114  continue;
115  switch ( eps[cnt]->d_type ) {
116  case DT_REG : {
117  __GIM_CLEAR( message , 256 , char );
118  currentlist = add( );
119  sprintf( TmpFileName , "%s%s" , dir_name , eps[cnt]->d_name );
120  if ( gim_file_manager->length( TmpFileName ) != 0 ) {
121  sprintf( message , "Adding regular file %s..." , eps[cnt]->d_name );
122  __GIM_CLEAR( currentlist->name , 256 , char );
123  strcpy( currentlist->name , eps[cnt]->d_name );
124  currentlist->size = gim_file_manager->length( TmpFileName );
125  }
126  gim_error->set( "gim_gkdp_obj::make_gkdp" , message );
127  break;
128  }
129  case DT_DIR :
130  case DT_LNK :
131  case DT_UNKNOWN :
132  case DT_BLK :
133  case DT_FIFO :
134  case DT_CHR :
135  case DT_SOCK :
136  default :
137  break;
138  }
139  }
140  }
141  if ( n <= 0 ) gim_error->set( GIM_ERROR_CRITICAL , "gim_gkdp_obj::make_gkdp" , "Couldn't open the directory" , __GIM_ERROR );
142  if ( n == 2 ) gim_error->set( GIM_ERROR_WARNING , "gim_gkdp_obj::make_gkdp" , "The directory is empty" , __GIM_NOT_OK );
143  if ( n > 2 ) {
144  char message [256];
145  sprintf( message , "File list maked : added %d members" , endlist->id + 1 );
146  gim_error->set( "gim_gkdp_obj::make_gkdp" , message );
147  }
148  header.members = endlist->id;
149  return endlist->id;
150 }

154 void gim_gkdp_obj::destroy( void ) {
155  _gim_gkp_flist * TmpList = NULL;
156  _gim_Uint32 c = 0;
157  if ( startlist != NULL ) {
158  gim_error->set( "gim_gkdp_obj::destroy" , "Destroy file list..." );
159  for( currentlist=startlist ; currentlist != NULL ; currentlist = TmpList , c++ ) {
160  char message[256];
161  if ( currentlist->id == 0 )
162  sprintf( message , "Releasing ROOT GKP::%s" , currentlist->name );
163  else
164  sprintf( message , "Releasing file GKP::%s" , currentlist->name );
165  gim_error->set( "gim_gkdp_obj::destroy" , message );
166  TmpList = currentlist->link;
167  if ( currentlist->buffer != NULL ) {
168  if ( gim_memory->Free( currentlist->buffer ) != __GIM_OK ){
169  gim_error->set( GIM_ERROR_CRITICAL , "gim_gkdp_obj::destroy" , "Some memory issue" , __GIM_ERROR );
170  }
171  else {
172  gim_error->set( "gim_gkdp_obj::destroy" , "Buffer leaved succesfully" );
173  }
174  }
175  if ( gim_memory->Free( currentlist ) != __GIM_OK )
176  gim_error->set( GIM_ERROR_CRITICAL , "gim_gkdp_obj::destroy" , "Some memory issue" , __GIM_ERROR );
177  }
178  if ( c ) {
179  char message[256];
180  sprintf( message , "Destroyed %d members" , c );
181  gim_error->set( "gim_gkdp_obj::destroy" , message );
182  }
183  startlist = NULL;
184  currentlist = NULL;
185  endlist = NULL;
186  gim_error->set( "gim_gkdp_obj::destroy" , "Done" );
187  }
188  else
189  gim_error->set( GIM_ERROR_WARNING , "gim_gkdp_obj::destroy" , "No memory list" , __GIM_ERROR );
190 }
