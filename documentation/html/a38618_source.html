113 _gim_flag gim_gkfp_obj::read_data( _gim_map map , _gim_gkp_flist * startlist ) {
114  _gim_gkp_flist * currentlist = NULL;
115  _gim_gkp_chunk chunk;
116  _gim_buffer data = NULL;
117  int c;
118  for( c = 0 , currentlist = startlist->link ; currentlist != NULL ; currentlist = currentlist->link , c++ ) {
119  if ( currentlist->type == REGFILE ) {
120  char message[256];
121  gim_interface->data->sou_size = gim_interface->data->sou_position = c + 1;
122  sprintf( message , "Reading GKP::%s " , currentlist->name );
123  gim_error->set( "gim_file_data_obj::data" , message );
124  __GIM_CLEAR( &chunk , 1 , chunk );
125  memcpy( &chunk , map , sizeof( chunk ) );
126  map += sizeof( chunk );
127  currentlist->handle = map;
128  data = (_gim_buffer)gim_memory->Alloc( currentlist->size , __GIM_MEM_FILE_DATA_R , __GIM_HIDE );
129  memcpy( data , map , currentlist->size );
130  map += currentlist->size;
131  sprintf( gim_interface->data->item_dst_name , "%s" , currentlist->name );
132  sprintf( gim_interface->data->item_dna_name , "Hashing" );
133  if ( internal_gim->get_state() == __GIM_TH_STOP ) return __GIM_NOT_OK;
134  if ( header.crypt == __GIM_SUN ) {
135  if ( Lexical.str_equal( chunk.id , gim_checksum->md5( data , currentlist->size ) ) == __GIM_NO ) {
136  char message[256];
137  sprintf( message , "GKP::%s checksum error" , currentlist->name );
138  gim_error->set( GIM_ERROR_CRITICAL , "gim_file_data_obj::data" , message , __GIM_ERROR );
139  }
140  }
141  if ( header.crypt == __GIM_FOG ) {
142  if ( Lexical.str_equal( chunk.xhash , gim_checksum->md5( data , currentlist->size ) ) == __GIM_NO ) {
143  char message[256];
144  sprintf( message , "GKP::%s checksum error" , currentlist->name );
145  gim_error->set( GIM_ERROR_CRITICAL , "gim_file_data_obj::data" , message , __GIM_ERROR );
146  }
147  }
148  gim_memory->Free( data );
149  }
150  }
151  return __GIM_OK;
152 }
