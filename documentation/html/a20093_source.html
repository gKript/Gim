42 void gim_log_obj::start ( void ) {
43  char message[256];
44  if ( ! initialized ) {
45 // gim_error->Set( "gim_log_obj::start" , "[" ,
46  if ( __GIM_LOG_FORCE == __GIM_YES ) {
47  log_fd = fopen ( __GIM_LOG_FILENAME , "wb" );
48  if ( log_fd == NULL ) {
49  gim_error->set( "gim_log_obj::start" , 6 );
50  }
51  else {
52  initialized = __GIM_YES;
53  enabled = __GIM_YES;
54  debug_timer.start_usec();
55  strcpy( format , "|$t|$p|$k_$f_|_$m" );
56  gim_error->set( "gim_log_obj::start" , "Start before configuration file due __GIM_LOG_FORCE" );
57  sprintf( message , "File log start on %s" , debug_timer.date() );
58  gim_error->set( "gim_log_obj::start" , message );
59  gim_error->set( "gim_log_obj::start" , "This log use a standard format [ |$t|$p|$k_$f_|_$m ] due __GIM_LOG_FORCE" );
60  }
61  }
62  if ( __GIM_LOG_FORCE == __GIM_NO ) {
63  if ( ! enabled ) {
64  if ( gim_conf->prsr_obj != NULL ) {
65  IF_EXIST_KEY_IN_CONF( "debug" , "f_debug" ) {
66  enabled = gim_conf->GetKeyFLAG( "debug" , "f_debug" );
67  }
68  IF_EXIST_KEY_IN_CONF( "DEBUG" , "debug_format" ) {
69  strcpy( format , gim_conf->GetKeySTR( "DEBUG" , "debug_format" ) );
70  }
71  else {
72  gim_error->set( GIM_ERROR_CRITICAL , "gim_log_obj::addline_format" , "Format NOT present in configuration file" , __GIM_ERROR );
73  }
74  }
75  else {
76  gim_error->set( GIM_ERROR_WARNING , "gim_log_obj::start" , "I cannot start the File debug in gim " , -1 );
77  gim_error->set( GIM_ERROR_WARNING , "gim_log_obj::start" , " 'cause the gim.conf was not loaded " , -1 );
78  }
79  }
80  if ( enabled ) {
81  log_fd = fopen ( gim_conf->GetKeySTR( "debug" , "f_debug_file" ) , "wb" );
82  if ( log_fd == NULL ) {
83  gim_error->set( "gim_log_obj::start" , 6 );
84  }
85  else {
86  initialized = __GIM_YES;
87  debug_timer.start_usec();
88  gim_error->set( "gim_log_obj::start" , 3 );
89  sprintf( message , "File log start on %s" , debug_timer.date() );
90  gim_error->set( "gim_log_obj::start" , message );
91  gim_error->set( "gim_log_obj::start" , gim_version() );
92  }
93  }
94  }
95  }
96  else {
97  gim_error->set( GIM_ERROR_WARNING , "gim_log_obj::start" , "The timer is already initialized" , __GIM_OK );
98  }
99 }

167 void gim_log_obj::addline_format( const char *t , const char *func , const char *mess , int val ) {
168  if ( enabled ) {
169  if ( initialized == __GIM_YES ) {
170  if ( log_fd != NULL ) {
171  }
172  else {
173  gim_error->set( "gim_log_obj::addline_format" , 7 );
174  }
175  char tmp[1024];
176  int c, len;
177  strcpy( tmp , "" );
178  for ( c = 0 , len = strlen( format ) ; c <= len ; c++ ) {
179  switch ( format[c] ) {
180  case '$' : {
181  switch ( format[++c] ) {
182  case 't' : {
183  sprintf( tmp , "%s%s" , tmp , debug_timer.get_usec() );
184  break;
185  }
186  case 'p' : {
187  sprintf( tmp , "%s%5d" , tmp , syscall(SYS_gettid) );
188  break;
189  }
190  case 'k' : {
191  if ( Lexical.str_equal( strdup( t ) , strdup( GIM_ERROR_FATAL ) ) == __GIM_YES )
192  sprintf( tmp , "%s%-4s" , tmp , __GIM_LOG_FATAL );
193  else
194  sprintf( tmp , "%s%-4s" , tmp , t );
195  break;
196  }
197  case 'f' : {
198  sprintf( tmp , "%s%s" , tmp , Lexical.string_trunc( (char *)func , 48 ) );
199  break;
200  }
201  case 'm' : {
202  sprintf( tmp , "%s%-45s" , tmp , mess );
203  break;
204  }
205  case 'v' : {
206  sprintf( tmp , "%s%3d" , tmp , val );
207  break;
208  }
209  case 'T' : {
210  sprintf( tmp , "%s%s" , tmp , (syscall(SYS_gettid) != getpid()) ? "TH" : " " );
211  break;
212  }
213  case '$' : {
214  sprintf( tmp , "%s$" , tmp );
215  break;
216  }
217  default : {
218  gim_error->set( GIM_ERROR_WARNING , "gim_log_obj::addline_format" , "Unknow variable in DEBUG::debug_format value" , __GIM_NOT_OK );
219  break;
220  }
221  }
222  }
223  case '_' : {
224  sprintf( tmp , "%s " , tmp );
225  break;
226  }
227  default : {
228  sprintf( tmp , "%s%c" , tmp , format[c] );
229  break;
230  }
231  }
232  }
233  fprintf( log_fd , "%s\n" , tmp );
234  fflush( log_fd );
235  }
236  }
