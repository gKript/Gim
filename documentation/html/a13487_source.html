24  gim_error->set( "gim_cnc_obj::set" , "Changeged a Feature in CNC kernel" );
25  if ( ( state == __GIM_ON ) || ( state == __GIM_OFF ) ) {
26  switch (feature_to_set)

28  case NOSKEMA : {

30  return __GIM_OK;

33  case VARIABLEAB : {
