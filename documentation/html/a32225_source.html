134 void * gim_memory_obj::Alloc ( size_t size , _gim_Uint32 id , _gim_flag hidden ) {
135  if ( virt_mem == __GIM_OFF ) {
136  if ( gim_interface != NULL )
137  gim_interface->data->mem_max_allocable = max_allocation;
138  if ( size > (size_t)abs( (int)max_allocation - allocated ) ) {
139  if ( max_allocation > 0 ) {
140  char message[256];
141  sprintf( message , "The demand for memory exceeds the max_allocation [ %s / %d ]" , metric_converter( size ) , ( max_allocation - allocated ) );
142  gim_error->set( GIM_ERROR_FATAL , "gim_memory_obj::alloc" , message , __GIM_ERROR );
143  return NULL;
144  }
145  }
146  }
147  gim_mem_list * tmplist = NULL;
148  if ( id != __GIM_MEM_OTHER ) {
149  caller = id;
150  }
151  tmplist = ( gim_mem_list * ) malloc ( sizeof ( gim_mem_list ) );
152  if ( tmplist == NULL )
153  gim_error->set( GIM_ERROR_FATAL , "gim_memory_obj::alloc", "Failed during a allocation of a new member of memory list" , __GIM_ERROR );
154  tmplist->data = ( void * ) malloc ( size );
155  if ( tmplist->data == NULL ) {
156  char message[256];
157  sprintf( message , "Memory allocation failed on a request of %s size from %s" , metric_converter( size ) , code_to_id( caller ) );
158  gim_error->set( GIM_ERROR_FATAL , "gim_memory_obj::alloc", message , __GIM_ERROR );
159  return NULL;
160  }
161  allocated += size;
162  if ( allocated > allocated_peek ) {
163  allocated_peek = allocated;
164  }
165  if ( gim_interface != NULL ) {
166  if ( gim_interface->data != NULL ) {
167  gim_interface->data->mem_allocated = allocated;
168  gim_interface->data->mem_ratio = (float)( (float)allocated / (float)max_allocation );
169  gim_interface->data->mem_allocated_peek = allocated_peek;
170  gim_interface->data->mem_allocated_peek_ratio = (float)( (float)allocated_peek / (float)max_allocation );
171  }
172  }
173  memset( tmplist->data , 0 , size );
174  tmplist->size = size;
175  tmplist->caller = caller;
176  tmplist->hide = hidden;
177  tmplist->lock = this->lock;
178  endlist->next = tmplist;
179  endlist = tmplist;
180  endlist->next = NULL;
181  if ( ( hidden != __GIM_HIDE ) || ( Hide == __GIM_NO ) ) {
182  char message[256];
183  sprintf( message , "Allocated [%s] index[%p] , Allocation id : %s" , metric_converter( size ) , endlist->data , code_to_id( caller ) );
184  if ( tmplist->caller == id_to_code( "__ID_COMMAND_ERROR" ) ) {
185  gim_error->set( GIM_ERROR_CRITICAL , "gim_memory_obj::alloc", message , __GIM_ERROR );
186  }
187  else {
188  gim_error->set( "gim_memory_obj::alloc" , message );
189  }
190  if ( virt_mem == __GIM_OFF ) {
191  sprintf( message , "Total memory allocated [%d] | Free memory for the application [%d]" , allocated , ( max_allocation - allocated ) );
192  gim_error->set( "gim_memory_obj::alloc" , message );
193  }
194  }
195  caller = __GIM_MEM_OTHER;
196  return tmplist->data;
197 }

257 _gim_flag gim_memory_obj::Free ( void * to_free ) {
258  _gim_flag found = __GIM_NO;
259  gim_mem_list * before = startlist;
260  char message[256];
261  if ( to_free == NULL )
262  gim_error->set( GIM_ERROR_FATAL , "gim_memory_obj::Free" , "Is not possible to free a NULL pointer" , __GIM_ERROR );
263  if ( startlist == NULL )
264  gim_error->set( "gim_memory_obj::Free" , 14 );
265  else {
266  for ( currentlist = startlist ; currentlist != NULL ; currentlist = currentlist->next ) {
267  if ( ( unsigned long long ) to_free == ( unsigned long long )currentlist->data ) {
268  found = __GIM_YES;
269  if ( currentlist->lock == __GIM_MEM_LOCK ) {
270  if ( currentlist->caller == __GIM_MEM_OTHER ) {
271  gim_error->set( GIM_ERROR_CRITICAL , "gim_memory_obj::Free", "CRITICAL : You are attempting to free a LOCKED frame." , __GIM_ERROR );
272  sprintf( message , " [%s] index[%p] , Allocation id : %s." , metric_converter( currentlist->size ) , currentlist->data , code_to_id( currentlist->caller ) );
273  gim_error->set( GIM_ERROR_CRITICAL , "gim_memory_obj::Free", message , __GIM_ERROR );
274  gim_error->set( GIM_ERROR_CRITICAL , "gim_memory_obj::Free", " You cannot free." , __GIM_ERROR );
275 // tmplist = ( gim_mem_list * ) malloc ( sizeof ( gim_mem_list ) );
276  return __GIM_NOT_OK;
277  }
278  }
279  if ( ( currentlist->hide != __GIM_HIDE ) || ( Hide == __GIM_NO ) ) {
280  sprintf( message , "Free [%s] index[%p] , Allocation id : %s" , metric_converter( currentlist->size ) , currentlist->data , code_to_id( currentlist->caller ) );
281  gim_error->set( "gim_memory_obj::Free", message );
282  }
283  free( currentlist->data );
284  allocated -= currentlist->size;
285  if ( ( currentlist->hide != __GIM_HIDE ) || ( Hide == __GIM_NO ) ) {
286  if ( virt_mem == __GIM_OFF ) {
287  sprintf( message , "Total memory allocated [%d] | Free memory for the application [%d]" , allocated , (_gim_Uint32)( max_allocation - allocated ) );
288  gim_error->set( "gim_memory_obj::Free" , message );
289  }
290  }
291  if ( gim_interface != NULL ) {
292  if ( gim_interface->data != NULL ) {
293  gim_interface->data->mem_allocated = allocated;
294  gim_interface->data->mem_ratio = (float)( (float)allocated / (float)max_allocation );
295  }
296  }
297  before->next = currentlist->next;
298  if ( currentlist->next == NULL )
299  endlist = before;
300  free( currentlist );
301  to_free = NULL;
302  return __GIM_OK;
303  }
304  else
305  before = currentlist;
306  }
307  if ( found == __GIM_NO ) {
308  sprintf( message , "Frame NOT found : index[%p]" , to_free );
309  gim_error->set( GIM_ERROR_CRITICAL , "gim_memory_obj::Free" , message , __GIM_ERROR );
310  return __GIM_NOT_OK;
311  }
312  }
313  return __GIM_NOT_OK;
314 }

325 _gim_Uint32 gim_memory_obj::release_memory_list( void ) {
326  gim_mem_list * Next;
327  _gim_Uint32 total = 0 , not_empty = 0 , total_size = 0 ;
328  gim_error->set( "gim_memory_obj::release_memory_list" , 16 );
329  for ( currentlist = startlist ; currentlist != NULL ; currentlist = Next ) {
330  Next = currentlist->next;
331  if ( currentlist->data != NULL ) {
332  switch (currentlist->caller) {
333  case __GIM_MEM_OTHER : {
334  char message[256];
335  sprintf( message , "Free [%s] index[%p] __GIM_MEM_OTHER" , metric_converter( currentlist->size ) , currentlist->data );
336  gim_error->set( GIM_ERROR_WARNING , "gim_memory_obj::release_memory_list", message , __GIM_ERROR );
337  total_size += currentlist->size;
338  not_empty++;
339  free( currentlist->data );
340  free( currentlist );
341  currentlist = NULL;
342  break;
343  }
344  default : {
345  char message[256];
346  sprintf( message , "Free [%s] index[%p] , Allocation id : %s" , metric_converter( currentlist->size ), currentlist->data , code_to_id( currentlist->caller ) );
347  gim_error->set( GIM_ERROR_WARNING , "gim_memory_obj::release_memory_list", message , __GIM_ERROR );
348  total_size += currentlist->size;
349  not_empty++;
350  free( currentlist->data );
351  free( currentlist );
352  currentlist = NULL;
353  break;
354  }
355  }
356  }
357  else {
358  if ( currentlist->caller == id_to_code( "__GIM_MEM_MEMORY_UP" ) ) {
359  total -= 1;
360  }
361  }
362  total++;
363  }
364  char message[256];
365  total_size += ( sizeof( gim_mem_list ) * total );
366  sprintf( message , "Total element [%d] Not empty element [%d] Deallocated in MBytes [ %s - %d ]" , total , not_empty , metric_converter( total_size ) , total_size );
367  gim_error->set( "gim_memory_obj::destroy_list", message );
368  currentlist = NULL;
369  endlist = NULL;
370  return total;
371 }
