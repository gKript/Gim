47 gim_dir_file_list * gim_directory_obj::add( void ) {
48  gim_dir_file_list * TmpList;
49  TmpList = (gim_dir_file_list *)gim_memory->Alloc( sizeof (gim_dir_file_list) , __GIM_MEM_DIRECTORY_FLIST_ADD , __GIM_HIDE );
50  if ( TmpList == NULL )
51  gim_error->set( GIM_ERROR_FATAL , "gim_directory_obj::add" , "Allocation of a new file list member failed" , __GIM_ERROR );
52  if ( ! startlist ) {
53  startlist = TmpList;
54  startlist->link = NULL;
55  endlist = TmpList;
56  TmpList->id = 0;
57  }
58  else {
59  endlist->link = TmpList;
60  TmpList->link = NULL ;
61  TmpList->id = endlist->id + 1;
62  endlist = TmpList;
63  }
64  return TmpList;
65 }

68  struct dirent ** eps;
69  int n , cnt;
70  char TmpFileName[1024];
71  char message[256];
72 // sprintf( message , "Scanning %s..." , dir_name );
73 // gim_error->set( "gim_directory_obj::make" , message );
74  n = scandir ( dir_name , &eps , NULL , alphasort );
75  if ( n ) {
76  if ( level == __GIM_ROOT ) {

88  struct stat tmp;

91  continue;
92  if ( ( *(eps[cnt]->d_name) == '.' ) && ( hide == __GIM_NO ) )
93  continue;

95  continue;

103  strcpy( currentlist->path , dir_name );
104  if ( gim_file_manager->length( TmpFileName ) != 0 ) {
105  currentlist->type = REGFILE;
106  currentlist->size = gim_file_manager->length( TmpFileName );
107 // sprintf( message , "Added regular file %s..." , eps[cnt]->d_name );
108  }
109  else {
110  currentlist->type = EMPTY;
111  currentlist->size = 0;
112 // sprintf( message , "Added empty regular file %s..." , eps[cnt]->d_name );
113  }
114  stat( TmpFileName , &( currentlist->stat ) );
115 // gim_error->set( "gim_directory_obj::make" , message );
116  break;
117  }
118  case DT_DIR : {
119  currentlist = add( );
120  sprintf( TmpFileName , "%s%s/" , dir_name , eps[cnt]->d_name );
121  __GIM_CLEAR( currentlist->name , 256 , char );
122  strcpy( currentlist->name , eps[cnt]->d_name );
123  strcpy( currentlist->path , dir_name );

131  }
132  case DT_LNK : {

141 */ break;
142  }

147  }

150  gim_error->set( "gim_directory_obj::make" , message );
151 */ break;
152  }
153  case DT_CHR : {
154 /* sprintf( message , "Skipped Character file %s..." , eps[cnt]->d_name );
155  gim_error->set( "gim_directory_obj::make" , message );
156 */ break;
157  }
158  case DT_SOCK : {
159 /* sprintf( message , "Skipped Socket file %s..." , eps[cnt]->d_name );
160  gim_error->set( "gim_directory_obj::make" , message );
161 */ break;
162  }
163  case DT_UNKNOWN : {
164 /* sprintf( message , "Skipped UNKNOWN file %s..." , eps[cnt]->d_name );
165  gim_error->set( "gim_directory_obj::make" , message );
166 */ break;
167  }
168  }
169  }
170  }
171  if ( n <= 0 )
172  gim_error->set( GIM_ERROR_WARNING , "gim_directory_obj::make" , "Couldn't open the directory" , __GIM_ERROR );
173 // if ( n == 2 )
174 // gim_error->set( "gim_directory_obj::make" , "The directory is empty" );
175  if ( ( n > 2 ) && ( level == __GIM_ROOT ) ) {
176  char message [256];
177  sprintf( message , "items are %-8d : %s" , endlist->id + 1 , dir_name );
178  gim_error->set( "gim_directory_obj::make" , message );
179  }
180  item = endlist->id;
181  return endlist->id;
182 }

204  gim_error->set( GIM_ERROR_CRITICAL , "gim_directory_obj::destroy" , "Some memory issue" , __GIM_ERROR );
205  }
206  if ( c ) {
207  char message[256];
208  sprintf( message , "Destroyed %d members" , c );
209  gim_error->set( "gim_directory_obj::destroy" , message );

211  startlist = NULL;

214  gim_error->set( "gim_directory_obj::destroy" , "Done" );
