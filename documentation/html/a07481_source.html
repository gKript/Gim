39 void gim_gkdp_obj::write_gkdp( _gim_handler * out ) {
40  char path[1024];
41  _gim_handler * in = NULL;
42  char * file = NULL , tname[1024];
43  _gim_gkp_chunk chunk;
44  if ( Lexical.str_equal( "ManualGKDP" , startlist->name ) != __GIM_YES ) {
45  strcpy( path , startlist->name );
46  for( currentlist = startlist->link ; currentlist != NULL ; currentlist = currentlist->link ) {
47  sprintf( tname , "%s%s" , path , currentlist->name ) ;
48  __GIM_CLEAR( &chunk , 1 , _gim_gkp_chunk );
49  if ( header.crypt == __GIM_YES )
50  in = gim_file_manager->open( tname , __GIM_FILE_POINTER , __GIM_READ );
51  if ( header.crypt == __GIM_NO )
52  in = gim_file_manager->open( tname , __GIM_FILE_MAPPING , __GIM_READ );
53  if ( in == NULL ) {
54  char message[256];
55  sprintf( message , "I cannot open file %s" , currentlist->name );
56  gim_error->set( GIM_ERROR_FATAL , "gim_file_data_obj::data" , message , __GIM_ERROR );
57  }
58  if ( header.crypt == __GIM_YES ) {
59  file = gim_file_manager->load( in );
60  if ( file == NULL )
61  gim_error->set( GIM_ERROR_FATAL , "gim_file_data_obj::data" , "I cannot load file" , __GIM_NOT_OK );
62  strcpy( chunk.id , gim_checksum->md5( file , in->size ) ) ;
63  crypt->set_secondary( currentlist->name );
64  crypt->advanced_password( file , in->size );
65  strcpy( chunk.xhash , gim_checksum->md5( file , in->size ) );
66  }
67  else {
68  strcpy( chunk.id , gim_checksum->md5( in->fm , in->size ) ) ;
69  }
70  char message[1024];
71  sprintf( message , "writing chunk id - %d" , currentlist->id );
72  gim_error->set( "gim_file_data_obj::data" , message );
73  gim_file_manager->flush( out , (char *)&chunk , sizeof( chunk ) );
74  __GIM_CLEAR( message , 1024 , char );
75  sprintf( message , "writing file - %s" , currentlist->name );
76  gim_error->set( "gim_file_data_obj::data" , message );
77  if ( header.crypt == __GIM_YES )
78  gim_file_manager->flush( out , file , in->size );
79  if ( header.crypt == __GIM_NO )
80  gim_file_manager->flush( out , in->fm , in->size );
81  gim_file_manager->close( in );
82  }
83  }
84  else {
85  gim_interface->clear();
86  for( currentlist = startlist->link ; currentlist != NULL ; currentlist = currentlist->link ) {
87  gim_interface->data->sou_ratio = (float)( ( (float)currentlist->id / (float)endlist->id ) / 100.00 );
88  strcpy( gim_interface->data->item_sou_name , currentlist->name );
89  __GIM_CLEAR( &chunk , 1 , _gim_gkp_chunk );
90  strcpy( chunk.id , gim_checksum->md5( currentlist->handle , currentlist->size ) ) ;
91  if ( header.crypt == __GIM_YES ) {
92  crypt->set_secondary( currentlist->name );
93  crypt->advanced_password( currentlist->handle , currentlist->size );
94  strcpy( chunk.xhash , gim_checksum->md5( currentlist->handle , currentlist->size ) );
95  }
96  char message[1024];
97  sprintf( message , "writing chunk id - %d" , currentlist->id );
98  gim_error->set( "gim_file_data_obj::data" , message );
99  gim_file_manager->flush( out , (char *)&chunk , sizeof( chunk ) );
100  __GIM_CLEAR( message , 1024 , char );
101  sprintf( message , "writing file - %s" , currentlist->name );
102  gim_error->set( "gim_file_data_obj::data" , message );
103  gim_file_manager->flush( out , currentlist->handle , currentlist->size );
104  }

109 void gim_gkdp_obj::read_gkdp( _gim_map map ) {
110  _gim_gkp_chunk chunk;
111  _gim_buffer data = NULL;
112  for( currentlist = startlist->link ; currentlist != NULL ; currentlist = currentlist->link ) {
113  char message[256];
114  sprintf( message , "Reading GKP::%s " , currentlist->name );
115  gim_error->set( "gim_file_data_obj::data" , message );
116  __GIM_CLEAR( &chunk , 1 , chunk );
117  memcpy( &chunk , map , sizeof( chunk ) );
118  map += sizeof( chunk );
119  currentlist->handle = map;
120  data = (_gim_buffer)gim_memory->Alloc( currentlist->size , __GIM_MEM_FILE_DATA_R , __GIM_HIDE );
121  memcpy( data , map , currentlist->size );
122  map += currentlist->size;
123  if ( header.crypt == __GIM_NO ) {
124  if ( Lexical.str_equal( chunk.id , gim_checksum->md5( data , currentlist->size ) ) == __GIM_NO ) {
125  char message[256];
126  sprintf( message , "GKP::%s checksum error" , currentlist->name );
127  gim_error->set( GIM_ERROR_CRITICAL , "gim_file_data_obj::data" , message , __GIM_ERROR );
128  }
129  }
130  if ( header.crypt == __GIM_YES ) {
131  if ( Lexical.str_equal( chunk.xhash , gim_checksum->md5( data , currentlist->size ) ) == __GIM_NO ) {
132  char message[256];
133  sprintf( message , "GKP::%s checksum error" , currentlist->name );
134  gim_error->set( GIM_ERROR_CRITICAL , "gim_file_data_obj::data" , message , __GIM_ERROR );
135  }
136  }
137  gim_memory->Free( data );
138  }
139 }
