58 void gim_sha1_obj::SHA1Transform(unsigned int state[5], unsigned char buffer[64]) {
59  unsigned long a, b, c, d, e;
60  typedef union {
61  unsigned char c[64];
62  unsigned long l[16];
63  } CHAR64LONG16;
64  CHAR64LONG16* block;
65 #ifdef SHA1HANDSOFF
66  static unsigned char workspace[64];
67  block = (CHAR64LONG16*)workspace;
68  memcpy(block, buffer, 64);
69 #else
70  block = (CHAR64LONG16*)buffer;
71 #endif
72  /* Copy context->state[] to working vars */
73  a = state[0];
74  b = state[1];
75  c = state[2];
76  d = state[3];
77  e = state[4];
78  /* 4 rounds of 20 operations each. Loop unrolled. */
79  R0(a,b,c,d,e, 0); R0(e,a,b,c,d, 1); R0(d,e,a,b,c, 2); R0(c,d,e,a,b, 3);

122 void gim_sha1_obj::SHA1Update(SHA1_CTX* context, unsigned char* data, unsigned int len) {
123  unsigned int i, j;
124  j = (context->count[0] >> 3) & 63;
125  if ((context->count[0] += len << 3) < (len << 3)) context->count[1]++;
126  context->count[1] += (len >> 29);
127  if ((j + len) > 63) {
128  memcpy(&context->buffer[j], data, (i = 64-j));
129  SHA1Transform(context->state, context->buffer);
130  for ( ; i + 63 < len; i += 64) {
131  SHA1Transform(context->state, &data[i]);
132  }
133  j = 0;
134  }
135  else i = 0;
136  memcpy(&context->buffer[j], &data[i], len - i);
137 }

140 void gim_sha1_obj::SHA1Final(unsigned char digest[20], SHA1_CTX* context) {
141  unsigned long i, j;
142  unsigned char finalcount[8];
143  for (i = 0; i < 8; i++) {
144  finalcount[i] = (unsigned char)((context->count[(i >= 4 ? 0 : 1)]
145  >> ((3-(i & 3)) * 8) ) & 255); /* Endian independent */
146  }
147  SHA1Update(context, (unsigned char *)"\200", 1);
148  while ((context->count[0] & 504) != 448) {
149  SHA1Update(context, (unsigned char *)"\0", 1);
150  }
151  SHA1Update(context, finalcount, 8); /* Should cause a SHA1Transform() */
152  for (i = 0; i < 20; i++) {
153  digest[i] = (unsigned char)
154  ((context->state[i>>2] >> ((3-(i & 3)) * 8) ) & 255);
155  }
156  /* Wipe variables */
157  i = j = 0;
158  memset(context->buffer, 0, 64);
159  memset(context->state, 0, 20);
160  memset(context->count, 0, 8);
161  memset(&finalcount, 0, 8);
162 #ifdef SHA1HANDSOFF /* make SHA1Transform overwrite it's own static vars */
163  SHA1Transform(context->state, context->buffer);
164 #endif
